## Rust

1. Install latest EMACS edition. Here we assume EMACS 30.1, which has built-in Tree-sitter and LSP support (Eglot). Use some LLM to help with the install for your linux distro.
2. Install `rust-analyzer` binary https://rust-analyzer.github.io/book/rust_analyzer_binary.html
   ```
   rustup component add rust-analyzer
   ```
3. Install `rust-src` binary https://rust-analyzer.github.io/book/installation.html
   ```
   rustup component add rust-src
   ```
4. Create config file and open in Emacs
   ```
   mkdir -p ~/.emacs.d && touch ~/.emacs.d/init.el
   emacs ~/.emacs.d/init.el
   ```
5. Copy the following configurations
    ```
    ;;; =====================================================================
    ;;; EMACS 30.1 — MODERN RUST CONFIGURATION
    ;;; Tree-sitter + Eglot + rust-analyzer
    ;;; =====================================================================
    
    ;;; ---------------------------------------------------------------
    ;;; 1. PACKAGE MANAGEMENT
    ;;; ---------------------------------------------------------------
    
    (require 'package)
    (add-to-list 'package-archives '("melpa" . "https://melpa.org/packages/") t)
    (package-initialize)
    
    (unless package-archive-contents
      (package-refresh-contents))
    
    (require 'use-package)
    (setq use-package-always-ensure t)
    
    
    ;;; ---------------------------------------------------------------
    ;;; 2. TREE-SITTER
    ;;; ---------------------------------------------------------------
    
    (setq treesit-language-source-alist
          '((rust "https://github.com/tree-sitter/tree-sitter-rust")))
    
    (unless (treesit-language-available-p 'rust)
      (condition-case err
          (treesit-install-language-grammar 'rust)
        (error (message "Tree-sitter: failed to install Rust grammar: %S" err))))
    
    
    ;;; ---------------------------------------------------------------
    ;;; 3. EGLOT (LSP)
    ;;; ---------------------------------------------------------------
    
    (use-package eglot
      :ensure nil  ; built-in since Emacs 29
      :commands eglot-ensure
      :bind (:map eglot-mode-map
                  ("C-c r"   . eglot-rename)
                  ("C-c a"   . eglot-code-actions)
                  ("C-c f"   . eglot-format-buffer))
      :custom
      (eglot-events-buffer-size 0)          ; disable events log for performance
      (eglot-autoshutdown t)                ; shut down LSP when last buffer closes
      (eglot-connect-timeout 30)
      :config
      (setq-default eglot-workspace-configuration
      '(:rust-analyzer
        (:checkOnSave (:command "clippy")
         :procMacro (:enable t)
         :cargo (:buildScripts (:enable t))))))
    
    
    ;;; ---------------------------------------------------------------
    ;;; 4. RUST (rust-ts-mode — built-in)
    ;;; ---------------------------------------------------------------
    
    (use-package rust-ts-mode
      :ensure nil
      :mode "\\.rs\\'"
      :hook ((rust-ts-mode . eglot-ensure)
            ;; Add buffer-local before-save-hook (-10 depth ensures it runs early, 't' makes it local)
             (rust-ts-mode . (lambda ()
                               (add-hook 'before-save-hook #'eglot-format-buffer -10 t))))
      :custom
      (treesit-font-lock-level 4))          ; maximum syntax highlighting
    
    ;;; ---------------------------------------------------------------
    ;;; 5. EDITOR QUALITY OF LIFE
    ;;; ---------------------------------------------------------------
    
    ;; Indentation
    (setq-default indent-tabs-mode nil
                  tab-width 4)
    
    ;; Visual aids
    (global-display-line-numbers-mode 1)
    (column-number-mode 1)
    
    ;; Persistence
    (save-place-mode 1)
    (recentf-mode 1)
    (savehist-mode 1)                       ; persist minibuffer history
    
    ;; Editing conveniences
    (electric-pair-mode 1)
    (delete-selection-mode 1)               ; typing replaces selected region
    
    ;; Silence yes/no prompts
    (setq use-short-answers t)
    
    ;; Centralize backup and auto-save files
    (let ((backup-dir (expand-file-name "backups/"   user-emacs-directory))
          (auto-dir   (expand-file-name "auto-saves/" user-emacs-directory)))
      (make-directory backup-dir t)
      (make-directory auto-dir t)
      (setq backup-directory-alist         `(("." . ,backup-dir))
            auto-save-file-name-transforms `((".*" ,auto-dir t))
            backup-by-copying t            ; safer: copy instead of rename
            kept-new-versions 6
            kept-old-versions 2
            delete-old-versions t
            version-control t))            ; numbered backups
    
    ;;; init.el ends here
    ```
    After saving the file, restart Emacs and open a Rust file (e.g., `main.rs`).

Command: `M-! cargo run`
